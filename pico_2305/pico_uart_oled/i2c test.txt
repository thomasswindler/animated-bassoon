int i2c_address_test_main() {
    stdio_init_all();
    i2c_init(i2c0, 100000);
    gpio_set_function(4, GPIO_FUNC_I2C);
    gpio_set_function(5, GPIO_FUNC_I2C);
    gpio_pull_up(4); gpio_pull_up(5);
    sleep_ms(100);
    uint8_t buf[] = {0x00, 0xAE};
    int ret = i2c_write_blocking(i2c0, 0x3C, buf, sizeof(buf), false);
    if (ret >= 0) printf("ACK from 0x3C\n"); else printf("No ACK from 0x3C\n");
    while (true) {
        int ret = i2c_write_blocking(i2c0, 0x3C, buf, sizeof(buf), false);
        printf("i2c_write_blocking returned %d\n", ret);
        sleep_ms(1000);
    }
}
